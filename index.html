<script>

const PAL=[

  {bg:'#EEEDFE',bd:'#8B83E0',tx:'#26215C'},

  {bg:'#D6F2E8',bd:'#1D9E75',tx:'#04342C'},

  {bg:'#DAEAF8',bd:'#378ADD',tx:'#042C53'},

  {bg:'#FDE8B0',bd:'#D4A017',tx:'#3A2200'},

  {bg:'#FAD8C8',bd:'#D85A30',tx:'#4A1B0C'},

  {bg:'#F5CCDC',bd:'#C94F7C',tx:'#4B1528'},

  {bg:'#E3E2DA',bd:'#888780',tx:'#2C2C2A'},

];


const NODE_H=32, ROW_GAP=50, NODE_PADX=14, FONT=12, CHAR_W=6.8, IPAD=12, MIN_W=72, MAX_C=22;

const ROW_H=NODE_H+ROW_GAP;


function lbl(n){return n.length>MAX_C?n.slice(0,MAX_C-1)+'…':n;}

function bw(n){return Math.max(MIN_W,Math.ceil(lbl(n).length*CHAR_W)+IPAD*2);}

function xe(s){return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');}


function buildTree(lines){

  const root={name:'__root__',children:[],level:-1};

  for(const raw of lines){

    const parts=raw.split('/').map(s=>s.trim()).filter(Boolean).slice(0,7);

    if(!parts.length)continue;

    let cur=root;

    for(let i=0;i<parts.length;i++){

      let ch=cur.children.find(c=>c.name===parts[i]);

      if(!ch){ch={name:parts[i],children:[],level:i};cur.children.push(ch);}

      cur=ch;

    }

  }

  return root;

}


function layout(root){

  const nodes=[],edges=[];

  function place(node,depth,parent){

    if(node.name==='__root__'){node.children.forEach(c=>place(c,0,null));return;}

    const w=bw(node.name);

    const nd={name:node.name,level:depth,w,h:NODE_H,x:0,y:depth*ROW_H,children:[]};

    node._nd=nd; nodes.push(nd);

    if(parent){parent.children.push(nd);edges.push({p:parent,c:nd});}

    node.children.forEach(ch=>place(ch,depth+1,nd));

  }

  place(root,0,null);


  function assignX(nd,ox){

    if(!nd.children.length){nd.x=ox+nd.w/2;return ox+nd.w+NODE_PADX;}

    let cx=ox;

    for(const ch of nd.children)cx=assignX(ch,cx);

    nd.x=(nd.children[0].x+nd.children[nd.children.length-1].x)/2;

    const le=nd.x-nd.w/2;

    if(le<ox){const sh=ox-le;shift(nd,sh);}

    return Math.max(cx,nd.x+nd.w/2+NODE_PADX);

  }

  function shift(nd,dx){nd.x+=dx;nd.children.forEach(ch=>shift(ch,dx));}

  let ox=0;

  nodes.filter(n=>n.level===0).forEach(r=>{ox=assignX(r,ox);});

  return{nodes,edges};

}


function toSVG(nodes,edges){

  if(!nodes.length)return'';

  const PAD=20;

  const maxX=Math.max(...nodes.map(n=>n.x+n.w/2));

  const maxY=Math.max(...nodes.map(n=>n.y+n.h));

  const W=maxX+PAD, H=maxY+PAD;

  let o=`<svg xmlns="http://www.w3.org/2000/svg" viewBox="${-PAD} ${-PAD} ${W+PAD} ${H+PAD}" width="${W+PAD}" height="${H+PAD}" style="display:block;overflow:visible">`;


  // connectors

  const byP=new Map();

  for(const e of edges){if(!byP.has(e.p))byP.set(e.p,[]);byP.get(e.p).push(e.c);}

  const lc='rgba(0,0,0,0.18)';

  for(const[p,chs]of byP){

    const px=p.x, py=p.y+NODE_H, midY=py+ROW_GAP/2;

    if(chs.length===1){

      o+=`<line x1="${px}" y1="${py}" x2="${chs[0].x}" y2="${chs[0].y}" stroke="${lc}" stroke-width="1"/>`;

    }else{

      o+=`<line x1="${px}" y1="${py}" x2="${px}" y2="${midY}" stroke="${lc}" stroke-width="1"/>`;

      const lx=Math.min(...chs.map(c=>c.x)), rx=Math.max(...chs.map(c=>c.x));

      o+=`<line x1="${lx}" y1="${midY}" x2="${rx}" y2="${midY}" stroke="${lc}" stroke-width="1"/>`;

      chs.forEach(ch=>{o+=`<line x1="${ch.x}" y1="${midY}" x2="${ch.x}" y2="${ch.y}" stroke="${lc}" stroke-width="1"/>`;});

    }

  }


  // nodes

  for(const n of nodes){

    const c=PAL[Math.min(n.level,6)];

    const lx=n.x-n.w/2;

    o+=`<rect x="${lx}" y="${n.y}" width="${n.w}" height="${NODE_H}" rx="6" fill="${c.bg}" stroke="${c.bd}" stroke-width="0.8"/>`;

    o+=`<text x="${n.x}" y="${n.y+NODE_H/2+1}" text-anchor="middle" dominant-baseline="central" font-family="Inter,system-ui,sans-serif" font-size="${FONT}" font-weight="500" fill="${c.tx}">${xe(lbl(n.name))}</text>`;

  }

  o+=`</svg>`;

  return o;

}


function renderChart(){

  const raw=document.getElementById('input-area').value;

  const lines=raw.split('\n').map(l=>l.trim()).filter(Boolean);

  document.getElementById('werr').style.display='none';

  if(!lines.length){showErr('Enter at least one path.');return;}

  const tree=buildTree(lines);

  if(!tree.children.length){showErr('No valid paths. Use / to separate levels.');return;}

  const{nodes,edges}=layout(tree);

  document.getElementById('chart-wrap').innerHTML=toSVG(nodes,edges);

}


function showErr(msg){const e=document.getElementById('werr');e.textContent=msg;e.style.display='block';}

  document.getElementById('werr').style.display='none';

}
